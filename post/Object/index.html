Go语言基础之面向对象编程

2019-09-04

尽管Go语言中没有封装、继承、多态这些概念，但同样通过别的方式实现这些特性：

  • 封装：通过方法实现
  • 继承：通过匿名字段实现
  • 多态：通过接口实现

继承\color{red}{继承}继承

匿名字段

type Person struct {
	id   int
	name string
}

type Student struct {
	Person // 匿名字段，那么默认Student就包含了Person的所有字段
	add    string
}

func main() {
	//1.0
	var s1 Student = Student{Person{1, "jiajia"}, "sz"}
	fmt.Printf("s1:%#v\n", s1)

	//自动推到类型
	s2 := Student{Person{2, "jiajia"}, "gz"}
	fmt.Printf("s2:%#v\n", s2)

	s3 := Student{Person{id: 3}, "bj"}
	fmt.Printf("s3:%#v\n", s3)

	//成员操作
	var s4 Student
	s4.Person = Person{4, "jiajia"}
	s4.add = "sh"
	fmt.Printf("s4:%#v\n", s4)
}

同名字段

//人
type Person struct {
	name string
	age  int
	sex  byte
}

//学生
type Student struct {
	Person //只有类型，没有名字，匿名字段，继承了Person的成员

func main() {

	fmt.Printf("s1 = %+v\n", s1)
	s1.Person.name = "jiajia"
	fmt.Printf("s1 = %+v\n", s1) //s1 = {Person:{name:jiajia age:0 sex:0} add:sz id:1 name:Go}
}

结构体指针类型匿名字段

type mystring string //自定义类型

//人
type Person struct {
	name string
	age  int
	sex  byte
}

//学生
type Student struct {
	*Person //指针类型------匿名字段
	int     //基础类型的匿名字段
	mystring
}

func main() {
	//1，0 方法一
	var s Student
	s.Person = &Person{"JIAJIA", 18, 'm'}
	s.int = 1
	s.mystring = "hello go"
	fmt.Printf("s = %+v\n", s) //s = {Person:0xc000004480 int:2 mystring:hello go}
	fmt.Println(s.name, s.age, s.sex)

	// 2.0 方法二
	var s2 Student
	s2.Person = new(Person)
	s2.name = "Go"
	s2.age = 18
	s2.sex = 'm'
	s2.int = 2
	s2.mystring = "hello go"
	fmt.Printf("s2 = %+v\n", s2)

	//3.0 方法三
	s3 := Student{&Person{name: "Go", age: 18, sex: 'm'}, 3, "hello go"}
	fmt.Printf("s3 = %+v\n", s3)
}

输出结果：

s = {Person:0xc000004480 int:1 mystring:hello go}
JIAJIA 18 109
s2 = {Person:0xc0000044e0 int:2 mystring:hello go}
s3 = {Person:0xc000004520 int:3 mystring:hello go}

封装\color{red}{封装}封装

方法

带有接收者的函数，我们称为方法(method)。 本质上，一个方法则是一个和特殊类型关联的函数。

//面向对象，方法：给某个类型绑定一个函数
type bigInt int

//a 叫做接收者，接收者就是传递的一个参数
func (a bigInt) Add(b bigInt) bigInt {
	return a + b
}

func main() {
	//定义一个变量
	var a bigInt = 2
	//调用方式：变量名.函数(所需参数)
	result := a.Add(3)
	fmt.Println("result = ", result)
}

结构体类型添加方法

type myint int

type Person struct {
	id   int
	name string
}

//打印
func (p Person) Printinfo() {
	fmt.Println(p)
}

//改变内容
func (p *Person) changeinfo(i int, n string) {
	p.id = i
	p.name = n
	fmt.Println("内部:", *p)
}

func main() {
	p1 := Person{1, "jiajia"}
	//调用打印的方法
	p1.Printinfo()

	//调用改变的方法
	(&p1).changeinfo(2, "hello") //取地址操作等价于p1.changeinfo(2, "hello")
	fmt.Println("外部:", p1)

输出结果：

{1 jiajia}
内部: {2 hello}
外部: {2 hello}

方法集

type Person struct {
	name string
	age  int
	sex  byte
}

//指针方法，非指针方法都能调用。

func (p Person) SetInfoValue() {
	fmt.Printf("SetInfoValue\n")
}

func (p *Person) SetInfoPointer() {
	fmt.Printf("SetInfoPointer\n")
}

func main() {
	//结构体变量是一个指针变量，它能够调用哪些方法，这些方法就是一个集合，简称方法集
	s1 := &Person{"jiajia", 18, 'm'}
	s1.SetInfoValue()
	(*s1).SetInfoValue() //(*s)转换成s后调用。等价于24行

	s2 := Person{"jiajia", 18, 'm'}
	s2.SetInfoPointer() // 内部先把s转换为&s再调用。相当于(&s).SetInfoPointer()
	(&s2).SetInfoPointer()
}

方法的重写

type Person struct {
	name string
	age  int
	sex  byte
}

//Person类型，实现了一个打印输出的方法
func (tmp *Person) PrintInfo() {
	fmt.Printf("name = %s,age = %d,sex = %c\n", tmp.name, tmp.age, tmp.sex)
}

//有个学生，继承了Person字段，成员和方法都继承了
type Student struct {
	Person //匿名字段
	id     int
	addr   string
}

//这个学生也有自己的一个方法，这个方法和Person的方法同名，这种方法叫做重写
func (stu Student) PrintInfo() {
	fmt.Printf("name = %s,age = %d,sex = %c,id = %d,addr = %s\n", stu.name, stu.age, stu.sex, stu.id, stu.addr)
}

func main() {
	s := &Student{Person{"jiajia", 18, 'm'}, 2, "sz"}
	//就近原则
	s.PrintInfo() //调用的就是Student的方法
	//显示调用即可调用为Person方法
	//调用的就是Person的方法
	s.Person.PrintInfo()
}

输出结果:

name = jiajia,age = 18,sex = m,id = 2,addr = sz
name = jiajia,age = 18,sex = m

方法值和方法表达式

type Person struct {
	name string
	age  int
	sex  byte
}

func (p Person) SetInfoValue() {
	fmt.Printf("SetInfoValue:%p,%v\n", &p, p)
}

func (p *Person) SetInfoPointer() {
	fmt.Printf("SetInfoPointer:%p,%v\n", p, p)
}

func main() {
	p := Person{"jiajia", 18, 'm'}

	//方法值,隐藏了接收者
	f := p.SetInfoPointer
	f()
	//方法表达式
	f1 := (Person).SetInfoValue
	f1(p) //显式的把接收者传递过去 ==============》等价于p.SetInfoValue()

	f2 := (*Person).SetInfoPointer
	f2(&p) //显式的把接收者传递过去 ==============》等价于p.SetInfoPointer()
}

多态\color{red}{多态}多态

接口的定义和实现

type Humaner interface {
	//方法，只有声明，没有实现。由别的类型(自定义类型)实现
	sayhi()
}

type Student struct {
	name string
	id   int
}

//实现学生的方法
func (stu *Student) sayhi() {
	fmt.Printf("stu:%s,%d\n", stu.name, stu.id)
}

type Teacher struct {
	addr string
	lg   string
}

//实现老师的方法
func (tea *Teacher) sayhi() {
	fmt.Printf("tea:%s,%s\n", tea.addr, tea.lg)
}

type Str string

//实现Str的方法
func (tmp *Str) sayhi() {
	fmt.Printf("hello %s\n", *tmp)
}

func main() {
	//定义接口类型的变量
	var i Humaner

	//只要实现了此接口方法的类型，那么这个类型的变量(接收者类型)就可以赋值给i

	stu := &Student{"jiajia", 2} //取Student的地址赋值给stu
	i = stu                      //stu赋值给i
	i.sayhi()                    //i是接口类型，可以调用方法

	tea := &Teacher{"SZ", "GO"}
	i = tea
	i.sayhi()

	var str Str = "jiajia"
	i = &str
	i.sayhi()

}

输出结果：

stu:jiajia,2
tea:SZ,GO
hello jiajia

多态的使用

type Humaner interface {
	//方法，只有声明，没有实现。由别的类型(自定义类型)实现
	sayhi()
}

type Student struct {
	name string
	id   int
}

//实现学生的方法
func (stu *Student) sayhi() {
	fmt.Printf("stu:%s,%d\n", stu.name, stu.id)
}

type Teacher struct {
	addr string
	lg   string
}

//实现老师的方法
func (tea *Teacher) sayhi() {
	fmt.Printf("tea:%s,%s\n", tea.addr, tea.lg)
}

type Str string

//实现Str的方法
func (tmp *Str) sayhi() {
	fmt.Printf("hello %s\n", *tmp)
}

//定义一个普通的函数，函数的参数为接口类型
//只有一个函数，可以有不同表现，多态
func myfunc(i Humaner) {
	i.sayhi()
}

func main() {
	stu := &Student{"jiajia", 2}
	tea := &Teacher{"SZ", "Go"}
	var str Str = "jiajia"
	//调用同一个函数，不同表现，多种形态
	myfunc(stu)
	myfunc(tea)
	myfunc(&str)
	//定义一个切片
	x := make([]Humaner, 3)
	x[0] = stu
	x[1] = tea
	x[2] = &str
	for _, i := range x {
		i.sayhi()
	}
}

接口的继承

type Humaner interface {
	sing()
}

type Personer interface {
	Humaner //匿名字段,继承了sing()
	run(length int)
}

type Student struct {
	name string
	id   int
}

func (stu *Student) sing() {
	fmt.Println("stu会唱歌")
}

func (stu *Student) run(length int) {
	fmt.Printf("stu会跑%d米\n", length)
}

func main() {
	var i Personer
	stu := &Student{"jiajia", 2}
	i = stu
	i.run(100)
	pfunc := i.sing //继承过来的方法，并且使用方法值隐藏接收者变量的传递
	pfunc()
}

空接口

空接口(interface{})不包含任何的方法，正因为如此，所有的类型都实现了空接口，因此空接口可以存储任意类型的数值。

func test(args ...interface{}) {
	fmt.Printf("我是接口类型:%T\n值为:%v", args, args)
}

func main() {
	//空接口，万能类型,能够保存任意类型的值
	var i interface{} = 1
	fmt.Println("i = ", i)          //i =  1
	fmt.Printf("i type is %T\n", i) //i type is int

	i = true
	fmt.Println("i = ", i)          //i =  true
	fmt.Printf("i type is %T\n", i) //i type is bool

	test("jiajia", 1, 2, 'm')
}

类型断言(类型查询)

反向知道这个变量存储的是什么类型的变量
if方法：

type Student struct {
	name string
	id   int
}

func main() {
	i := make([]interface{}, 3)
	i[0] = 1
	i[1] = "hello go"
	i[2] = Student{"jiajia", 2}

	//类型查询，类型断言
	//第一个返回的是下标，第二个返回的是值
	for index, data := range i {
		//第一个返回的是值，第二个返回的是判断结果的真假
		if value, ok := data.(int); ok == true {
			fmt.Printf("i[%d]的类型为int,值为%d\n", index, value)
		} else if value, ok := data.(string); ok == true {
			fmt.Printf("i[%d]的类型为string,值为%s\n", index, value)
		} else if value, ok := data.(Student); ok == true {
			fmt.Printf("i[%d]Student,值为name = %s,id = %d\n", index, value.name, value.id)
		}
	}
}

switch方法：

type Student struct {
	name string
	id   int
}

func main() {
	i := make([]interface{}, 3)
	i[0] = 1
	i[1] = "hello go"
	i[2] = Student{"jiajia", 2}
	for index, data := range i {
		//需要注意的是data.(type)
		switch value := data.(type) {
		case int:
			fmt.Printf("i[%d]的类型为int,值为%d\n", index, value)
		case string:
			fmt.Printf("i[%d]的类型为string,值为%s\n", index, value)
		case Student:
			fmt.Printf("i[%d]Student,值为name = %s,id = %d\n", index, value.name, value.id)
		}
	}
}
