Go语言基础之结构体

2019-09-04

Go语言提供了一种自定义数据类型，可以封装多个基本数据类型，这种数据类型叫结构体，英文名称struct。

结构体的定义\color{red}{结构体的定义}结构体的定义

type 类型名 struct {
    字段名 字段类型
    字段名 字段类型
    …
}
  • 类型名：标识自定义结构体的名称，在同一个包内不能重复。
  • 字段名：表示结构体字段名。结构体中的字段名必须唯一。
  • 字段类型：表示结构体字段的具体类型。

举个例子，定义一个Student(学生)结构体，代码如下

type Student struct {
	id   int
	age  int
	name string
	add  string
}

结构体实例化\color{red}{结构体实例化}结构体实例化

只有当结构体实例化时，才会真正地分配内存。也就是必须实例化后才能使用结构体的字段。格式如下：
var 结构体实例 结构体类型

基础类型结构体实例化

type Student struct {
	id   int
	age  int
	name string
	add  string
}

func main() {
	//顺序初始化
	var s1 Student = Student{1, 18, "jiajia", "sz"}
	fmt.Printf("%#v\n", s1) //main.Student{id:1, age:18, name:"jiajia", add:"sz"}

	//部分初始化
	var s2 Student = Student{
		id:   2,
		name: "jiajia",
	}
	fmt.Printf("%#v", s2) //main.Student{id:2, age:0, name:"jiajia", add:""}
}

也可以通过"."点操作来初始化

type Student struct {
	id   int
	age  int
	name string
	add  string
}

func main() {
	//定义一个结构体变量
	var s Student
	//操作成员使用(.)操作
	s.name = "jiajia"
	s.age = 18

	fmt.Printf("%#v", s) //main.Student{id:0, age:18, name:"jiajia", add:""}
}

指针类型结构体实例化

type Student struct {
	id   int
	age  int
	name string
	add  string
}

func main() {
	//和08同样效果
	var s1 *Student = &Student{1, 18, "jiajia", "sz"}
	fmt.Printf("%#v\n", *s1) //main.Student{id:1, age:18, name:"jiajia", add:"sz"}

	//部分初始化
	var s2 *Student = &Student{
		id:   2,
		name: "jiajia",
		age:  19,
		add:  "gz",
	}
	fmt.Printf("%#v\n", *s2) //main.Student{id:2, age:19, name:"jiajia", add:"gz"}
}

也可以这样操作，注意3.0，代码如下：

//Student 是一个结构体
type Student struct {
	id   int
	age  int
	name string
	add  string
}

func main() {
	//1.0
	//定义一个结构体变量
	var s Student
	//定义一个结构体的指针变量，保存s的值
	var p1 *Student
	p1 = &s
	p1.add = "sz"
	p1.name = "jiajia"
	p1.age = 18
	fmt.Printf("s = %#v\n", s)
	fmt.Printf("*p1 = %#v\n", *p1)

	//2.0
	//通过new申请一个结构体
	p2 := new(Student)
	p2.age = 19
	p2.name = "jiajia"
	p2.add = "gz"
	fmt.Printf("*p2 = %#v\n", *p2)

	p3 := &Student{
		id:   3,
		age:  20,
		name: "jiajia",
		add:  "bj",
	}
	fmt.Printf("*p3 = %#v\n", *p3)
}

输出结果：

s = main.Student{id:0, age:18, name:"jiajia", add:"sz"}
*p1 = main.Student{id:0, age:18, name:"jiajia", add:"sz"}
*p2 = main.Student{id:0, age:19, name:"jiajia", add:"gz"}
*p3 = main.Student{id:3, age:20, name:"jiajia", add:"bj"}

结构体作为函数参数传递\color{red}{结构体作为函数参数传递}结构体作为函数参数传递

值传递

//Student 是一个结构体
type Student struct {
	id   int
	age  int
	name string
	add  string
}

func test(s1 Student) {
	s1.age = 20
	fmt.Println("test:s1.age = ", s1.age)
}

func main() {
	s1 := Student{1, 18, "jiajia", "sz"}
	test(s1)
	fmt.Println("main:s1.age = ", s1.age)
}

输出结果：

test:s1.age =  20
main:s1.age =  18

地址传递

type Student struct {
	id   int
	name string
}

func test(s1 *Student) {
	s1.id = 2
	fmt.Printf("test:%#v\n", *s1)
}

func main() {
	s1 := &Student{1, "jiajia"}
	test(s1)
	fmt.Printf("main:%#v\n", *s1)
}

输出结果：

test:main.Student{id:2, name:"jiajia"}
main:main.Student{id:2, name:"jiajia"}

结构体比较\color{red}{结构体比较}结构体比较

//Student 是一个结构体
type Student struct {
	id   int
	age  int
	name string
	add  string
}

func main() {
	s1 := Student{1, 18, "jiajia", "sz"}
	s2 := Student{1, 18, "jiajia", "sz"}
	s3 := Student{3, 18, "jiajia", "sz"}

	//比较
	fmt.Println("s1 == s2", s1 == s2) //s1 == s2 true
	fmt.Println("s1 == s3", s1 == s3) //s1 == s3 false

	//赋值
	var tmp Student
	tmp = s3
	fmt.Println("tmp = ", tmp)          //tmp =  {3 18 jiajia sz}
	fmt.Println("tmp == s3", tmp == s3) //tmp == s3 true
}

结构体内存布局\color{red}{结构体内存布局}结构体内存布局

type test struct {
	a int8
	b int8
	c int8
	d int8
}

func main() {
	n := test{
		1, 2, 3, 4,
	}
	fmt.Printf("n.a %p\n", &n.a)
	fmt.Printf("n.b %p\n", &n.b)
	fmt.Printf("n.c %p\n", &n.c)
	fmt.Printf("n.d %p\n", &n.d)
}

输出结果：

n.a 0xc000054058
n.b 0xc000054059
n.c 0xc00005405a
n.d 0xc00005405b
