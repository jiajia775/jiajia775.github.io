Go语言基础之文件操作

2019-09-05

Go语言中文件读写的相关操作。

打开和关闭文件\color{red}{打开和关闭文件}打开和关闭文件

os.Open()函数能够打开一个文件，返回一个* File和一个err。对得到的文件实例调用close()方法能够关闭文件。后面讲到的OpenFile()可以写入文件。

package main

import "os"

func main() {
	// 只读方式打开当前目录下的xxx.txt文件
	file,err:=os.Open("xxx.txt")
	if err:= nil{
		fmt.Println(err)
		return
	}
	//为了防止文件忘记关闭，我们通常使用defer注册文件关闭语句。
	defer file.Close()
}

读取文件\color{red}{读取文件}读取文件

基本使用

func main() {
	// 只读方式打开当前目录下的xxx.txt文件
	file, err := os.Open("xxx.txt")
	if err != nil {
		fmt.Println(err)
		return
	}
	//为了防止文件忘记关闭，我们通常使用defer注册文件关闭语句。
	defer file.Close()

	//读取文件
	var tmp = make([]byte, 128)
	n, err := file.Read(tmp)
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Printf("读取了%d个字节\n", n)
	fmt.Println(string(tmp[:n]))
}

循环读取

import (
	"fmt"
	"io"
	"os"
)

func main() {
	// 只读方式打开当前目录下的xxx.txt文件
	file, err := os.Open("xxx.txt")
	if err != nil {
		fmt.Println(err)
		return
	}
	//为了防止文件忘记关闭，我们通常使用defer注册文件关闭语句。
	defer file.Close()

	//循环读取文件
	for {
		var tmp = make([]byte, 128)//开辟一个128字节的slice作为缓冲
		//Read返回的是一个字节数和一个错误
		n, err := file.Read(tmp)
		if err == io.EOF {
			//把当前读了多少个字节的数据打印出来，然后退出
			fmt.Println(string(tmp[:n]))
			return
		}
		if err != nil {
			fmt.Println(err)
			return
		}
		fmt.Printf("读取了%d个字节\n", n)
		fmt.Println(string(tmp[:n]))
	}
}

bufio循环读取文件

bufio是在file的基础上封装了一层API，支持更多的功能。

package main

import (
	"bufio"
	"fmt"
	"io"
	"os"
)

func main() {
	f, err := os.Open("./xxx.txt")
	if err != nil {
		fmt.Println(err)
		return
	}
	defer f.Close()

	//循环读取文件
	//新建一个缓冲区，返回的是一个读的对象
	reader := bufio.NewReader(f)
	for {
		//
		line, err := reader.ReadString('\n')
		if err == io.EOF {
			fmt.Print(line)
			return
		}
		if err != nil {
			return
		}
		fmt.Print(line)
	}
}

ioutil读取整个文件

io/ioutil包的ReadFile方法能够读取完整的文件，只需要将文件名作为参数传入。

package main

import (
	"fmt"
	"io/ioutil"
)

func main() {

	//读取文件
	content, err := ioutil.ReadFile("./xxx.txt")
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Println(string(content))
}

写入文件\color{red}{写入文件}写入文件

os.OpenFile()函数能够以指定模式打开文件，从而实现文件写入。

func OpenFile(name string, flag int, perm FileMode) (*File, error) {
	...
}
  • name：要打开的文件名
  • perm：文件的权限，一个八进制数，r（读）04，w（写）02，x（执行）01。
  • flag：文件打开的模式，具体如下表

Write和WriteString

package main

import (
	"fmt"
	"os"
)

func main() {
	//写入文件
	fileObj, err := os.OpenFile("yyy.txt", os.O_CREATE|os.O_WRONLY|os.O_TRUNC, 0666)
	if err != nil {
		fmt.Println(err)
		return
	}
	defer fileObj.Close()
	fileObj.WriteString("hello go\n") //直接写入字符串数据
	str := "hello jiajia"
	fileObj.Write([]byte(str)) //写入字节切片数据
}

bufio.NewWriter

package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	//写入文件
	fileObj, err := os.OpenFile("zzz.txt", os.O_CREATE|os.O_WRONLY|os.O_TRUNC, 0666)
	if err != nil {
		fmt.Println(err)
		return
	}
	defer fileObj.Close()

	writer := bufio.NewWriter(fileObj)
	for i := 0; i < 10; i++ {
		writer.WriteString("hello jiajia\n") //将数据先写入缓冲区
	}
	writer.Flush() //将缓存区中的内容写入文件
}

ioutil.WriteFile

package main

import (
	"fmt"
	"io/ioutil"
)

func main() {
	str := "hello jiajia"
	err := ioutil.WriteFile("./aaa.txt", []byte(str), 0666)
	if err != nil {
		fmt.Println("error:", err)
		return
	}
}
